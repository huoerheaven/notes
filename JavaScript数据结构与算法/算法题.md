```js
/**
 * 旋转数组 k 步 --- 使用concat (更优)
 * @param {*} arr 
 * @param {*} k 
 */
function rotate1(arr,k){
    let len=arr.length;
    if(!k || len===0) return arr;
    const step = Math.abs(k % len);

    const arr1 = arr.slice(-step);
    const arr2 = arr.slice(0,len-step);

    return arr1.concat(arr2);
}
```

```js
/**
 * 二分查找（循环）
 * @param {*} arr 
 * @param {*} target 
 * @returns index
 */
function binarySearch1(arr,target){
    const length = arr.length;
    if(length==0) return -1;
    let startIndex = 0;
    let endIndex=length-1;
    while(startIndex<=endIndex){
        let midIndex = Math.floor((startIndex+endIndex)/2);
        let midValue = arr[midIndex];
        if(target<midValue){
            //目标值较小，则继续在左侧查找
            endIndex = midIndex-1;
        }else if(target>midValue){
            //目标值较大，则继续在右侧查找
            startIndex=midIndex+1;
        }else{
            return midIndex;
        }
    }
    return -1;
 
}

/**
 * 二分查找（递归）
 * @param {*} arr 
 * @param {*} target 
 * @param {*} startIndex 
 * @param {*} endIndex 
 * @returns index
 */
function binarySearch2(arr,target,startIndex=0,endIndex=arr.length-1){
    const length = arr.length;
    if(length==0) return -1;
    if(startIndex>endIndex) return -1;

    //中间位置
    let midIndex = Math.floor((startIndex+endIndex)/2);
    let midValue = arr[midIndex];

    if(target<midValue){
        //目标值较小，则继续在左侧查找
        return binarySearch2(arr,target,startIndex,midIndex-1);
    }else if(target>midValue){
        //目标值较大，则继续在右侧查找
        return binarySearch2(arr,target,midIndex+1,endIndex);
    }else{
        return midIndex;
    }
}

/**
 * 二分查找（递归）
 * @param {*} arr 
 * @param {*} target 
 * @returns index
 */
function binarySearch3(arr,target){
    const length = arr.length;
    if(length==0) return -1;
    let startIndex = 0;
    let endIndex=length-1;
    function rec(startIndex,endIndex){
        if(startIndex>endIndex) return -1;
        //中间位置
        let midIndex = Math.floor((startIndex+endIndex)/2);
        let midValue = arr[midIndex];
        if(target<midValue){
            //目标值较小，则继续在左侧查找
            return rec(startIndex,midIndex-1);
        }else if(target>midValue){
            //目标值较大，则继续在右侧查找
            return rec(midIndex+1,endIndex);
        }else{
            return midIndex;
        }
    }
    return rec(startIndex,endIndex);
}
```

```js
 * 斐波那契数列（循环）
 * @param {*} n

 */
 function fibonacci(n){
    if(n<=0) return 0;
    if(n===1) return 1;
    let n1=1;//记录n-1的结果
    let n2=0//记录n-2的结果
    let res=0;
    for(let i=2;i<=n;i++){
        res=n1+n2;
        //记录中间结果
        n2=n1;
        n1=res;
    }
    return res;
}
```

```js
/**
 * 移动0到数组末尾（双指针）
 * @param {*} arr  number arr
 */
function moveZero(arr){
    const length=arr.length;
    if(length===0) return;

    let i;
    let j=-1;//指向第一个0
    for(i=0;i<length;i++){
        if(arr[i]===0 && j<0) {
            j=i;//指向第一个0
        }
        if(arr[i]!==0 && j>=0){
            const temp=arr[i];
            arr[i]=arr[j];
            arr[j]=temp;
            j++;
        }
    }
}
```

```js
/**
 * 查询1-max之间的所有对称数（数组反转）
 * @param {*} max  最大值
 * @returns 
 */
function fundPalindromeNumbers1(max){
    const res=[];
    if(max<1) return res;

    for(let i=1;i<=max;i++){
        //转化成字符串，在转换为数组，再反转，再转换成字符串，进行比较
        const s=i.toString();
        if(s===s.split("").reverse().join("")){
            res.push(i)
        }
    }
    return res;                        
}

/**
 * 查询1-max之间的所有对称数（字符串前后比较）
 * @param {*} max  最大值
 * @returns 
 */
 function fundPalindromeNumbers2(max){
    const res=[];
    if(max<1) return res;

    for(let i=1;i<=max;i++){
        const s=i.toString();
        const length=s.length;
        //字符串头尾比较
        let startIndex = 0;//字符串开始
        let endIndex=length-1;//字符串结束
        let flag=true;
        while(startIndex<endIndex){
            if(s[startIndex]!==s[endIndex]){
                flag=false;
                break;
            }else{
                //继续比较
                startIndex++;
                endIndex--;
            
            }
        }
        if(flag) res.push(i);

    }
    return res;                        
}

/**
 * 查询1-max之间的所有对称数（生成翻转数）
 * @param {*} max  最大值
 * @returns 
 */
 function fundPalindromeNumbers3(max){
    const res=[];
    if(max<1) return res;

    for(let i=1;i<=max;i++){
        let n=i;
        let rev=0;//存储翻转数

        //生成翻转数
        while(n>0){
            rev=rev*10+n%10;
            n=Math.floor(n/10);
        }
        if(i===rev) res.push(i);

    }
    return res;                        
}
```

##### 7. 用链表实现队列
```js
/**
 * @description 用链表实现队列
 */
class MyQueue{
    #head=null;
    #tail=null;
    #len=0;
    /**
     * 入队，在tail位置
     * @param {*} n number
     */
    add(n){
        const newNode={value:n,next:null};
        //处理head
        if(this.#head===null) this.#head=newNode;
        //处理tail
        let tailNode = this.#tail;
        if(tailNode) tailNode.next = newNode;
        this.#tail=newNode;
        //记录长度
        this.#len++;
    }
    /**
     * 出队，在head位置
     */
    delete(){
        const headNode = this.#head;
        if(headNode===null) return null;
        if(this.#len<=0) return null;
        //取值
        const value = headNode.value;
        //处理head
        this.#head=headNode.next;
        //记录长度
        this.#len--;
        return value;

    }
    get length(){
        //length要单独存储，不能遍历链表来实现（否则时间复杂度太高O(n)）
        return this.#len;
    }
}
```

##### 8. 反转单向链表

```js
/**
 * 反转单向链表
 * @param {*} listNode 
 * @returns 
 */
function reverseLinkList(listNode){
    let prevNode = undefined;
    let currNode = undefined;
    let nextNode = listNode;
    while(nextNode){
        //第一个元素，删掉next,防止循环引用
        if(!prevNode && currNode){
            delete currNode.next;
        }
        //反转指针
        if(currNode && prevNode){
            currNode.next =prevNode;
        }

        //整体向后移动指针
        prevNode = currNode;
        currNode = nextNode;
        nextNode = nextNode.next;
    }
    //最后一个的补充：当nextNode为空时，此时currNode尚未设置next
    currNode.next = prevNode;
    return currNode;
}

/**
 * 根据数组创建单向链表
 * @param {*} arr  number array
 */
function createLinkList(arr){
    const length = arr.length;
    if(length===0) throw new Error("arr is empty");
    let currNode = {value:arr[length-1]};
    if(length===1) return currNode;
    for(let i=length-2;i>=0;i--){
       currNode={
        value:arr[i],
        next:currNode
       } 
    }
    return currNode;
}

//test
let linkList = createLinkList([1,2,3,4,5]);
console.log(reverseLinkList(linkList));
```

##### 9. 查找和为n的两个数

```js
/**
 * 查找和为n的两个数(双指针)
 * 时间复杂度O(n)
 * @param {*} arr 递增数组
 * @param {*} n  
 * @returns  []
 */
function findTwoNumbers(arr,n){
    let res=[];
    let length=arr.length;
    if(length===0) return res;
    let i=0;//头（指针）
    let j=length-1;//尾（指针）
    while(i<j){
        let n1=arr[i];
        let n2=arr[j];
        let sum=n1+n2;
        if(sum>n){
            //sum大于n,则j要向前移动
            j--;
        }else if(sum<n){
            //sum小于n,则i要向后移动
            i++;
        }else{
            res.push(n1);
            res.push(n2);
            break;
        }
    }
    return res;
}

//test 
const arr=[1,2,4,7,11,15];
console.info(findTwoNumbers(arr,15));
```

##### 10. 摩尔投票算法
```js
/**
 * 描述：（letCode 169）
 * 给定一个大小为 n 的数组 nums ，返回其中的多数元素。多数元素是指在数组中出现次数 大于 ⌊ n/2 ⌋ 的元素。
 * 你可以假设数组是非空的，并且给定的数组总是存在多数元素。
 * 
 * 示例：
 * 输入：nums = [2,2,1,1,1,2,2]
 * 输出：2
 * @param {number[]} nums 
 * @returns  {number}
 */

 var majorityElement = function(nums) {
    if(!Array.isArray(nums)) return false;
    let major = nums[0];
    let count = 1;
    for(let i=1;i<nums.length;i++){
        if(major===nums[i]){
            count++;
        }else{
            count--;
            if(count===0) major = nums[i+1];
        }
        
    }
    return major;
};
```
