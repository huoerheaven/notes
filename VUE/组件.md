#### 单向数据流的理解
1. 父组件向子组件传值时，可以简化写法
```
//v-bind="params"
//:a="params.a" :b="params.b" :c="params.c" :d="params.d"
const app=Vue.createApp({
  data(){
    return {
      params:{
        a:11,
        b:22,
        c:33,
        d:44
      },
      a:11,
      b:22,
      c:33,
      d:44
    }
  },
  template:`
  <div><test :a="a" :b="b" :c="c" :d="d"/></div>
  //等价于
  <div><test v-bind="params" /></div>
  `
})
app.component("test",{
  props:{a,b,c,d},
  template:`<div>{{a}}-{{b}}-{{c}}-{{d}}</div>`
})
```
