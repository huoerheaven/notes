/**
 * 数组深度扁平化 递归 concat+push
 * @param {*} arr 
 * @returns 
 */
function flat2(arr){
    let res=[];
    for(let i=0;i<arr.length;i++){
        if(Array.isArray(arr[i])){
            res=res.concat(flat2(arr[i]))
        }else{
            res.push(arr[i]);
        }
    }
    return res;
}
